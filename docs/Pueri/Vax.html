def parse_notes
  r = ''
  max = @notes.size
  p = Pastel.new

  @notes.each_with_index do |line, k|
    pre = ' ' * (max - (k + 1))
    pre += '*' * (k.to_i + 1)
    r += "\n#{p.cyan(pre)} #{line}"
  end

  r
end
